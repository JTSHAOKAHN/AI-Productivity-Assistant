import os
from pathlib import Path
from dotenv import load_dotenv
from google import genai

# =========================
# LOAD .ENV SAFELY (IMPORTANT)
# =========================
BASE_DIR = Path(__file__).resolve().parent
env_path = BASE_DIR / ".env"
load_dotenv(dotenv_path=env_path)

# =========================
# GET API KEY
# =========================
api_key = os.getenv("GEMINI_API_KEY")

if not api_key:
    raise Exception("Gemini API key not found. Check your .env file.")

# =========================
# CONFIGURE NEW GEMINI SDK
# =========================
client = genai.Client(api_key=api_key)


# =========================
# AI FUNCTION
# =========================
def get_ai_response(prompt):
    response = client.models.generate_content(
        model="gemini-2.5-flash",
        contents=prompt
    )
    return response.text


# =========================
# LOAD PROMPTS
# =========================
def load_prompt(file_name, user_input):
    file_path = BASE_DIR / "prompts" / file_name

    if not file_path.exists():
        raise FileNotFoundError(f"Prompt file not found: {file_path}")

    with open(file_path, "r", encoding="utf-8") as file:
        template = file.read()

    return template + "\n\n" + user_input


# =========================
# MAIN MENU
# =========================
def main():
    while True:
        print("\n==============================")
        print(" AI Productivity Assistant ")
        print("==============================")

        print("\n1. Generate Email")
        print("2. Summarize Meeting")
        print("3. Plan Tasks")
        print("4. Research Assistant")
        print("5. Exit")

        choice = input("\nSelect an option: ")

        try:
            if choice == "1":
                user_input = input("Describe the email: ")
                prompt = load_prompt("email_prompt.txt", user_input)
                print("\nGenerating email...\n")
                print(get_ai_response(prompt))

            elif choice == "2":
                user_input = input("Paste meeting notes: ")
                prompt = load_prompt("meeting_summary_prompt.txt", user_input)
                print("\nSummarizing...\n")
                print(get_ai_response(prompt))

            elif choice == "3":
                user_input = input("Enter tasks: ")
                prompt = load_prompt("task_planner_prompt.txt", user_input)
                print("\nPlanning tasks...\n")
                print(get_ai_response(prompt))

            elif choice == "4":
                user_input = input("Enter topic: ")
                prompt = load_prompt("research_prompt.txt", user_input)
                print("\nResearching...\n")
                print(get_ai_response(prompt))

            elif choice == "5":
                print("Goodbye!")
                break

            else:
                print("Invalid option. Try again.")

        except Exception as e:
            print(f"\n❌ Error: {e}")


if __name__ == "__main__":
    main()
